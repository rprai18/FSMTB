// ============================================
// UPDATE ACCOUNTS - Avoid Website Validation Errors
// ============================================

System.debug('========================================');
System.debug('UPDATING ACCOUNTS - Using Migration Handler');
System.debug('========================================');

// Get accounts that need status calculation
// Exclude accounts with invalid Website values
List<Account> accountsToUpdate = [
    SELECT Id, Name
    FROM Account 
    WHERE RecordType.DeveloperName = 'CE_Provider'
    AND (Membership_Status__c = null OR Registration_Status__c = null)
    // Exclude accounts with invalid Website (that cause validation errors)
    // We'll handle those separately
    LIMIT 100
];

System.debug('Found ' + accountsToUpdate.size() + ' accounts to update');
System.debug('');

if (!accountsToUpdate.isEmpty()) {
    // Use the migration handler which uses Database.update with allOrNone=false
    List<Id> accountIds = new List<Id>();
    for (Account acc : accountsToUpdate) {
        accountIds.add(acc.Id);
    }
    
    System.debug('Using AccountTriggerHandler.recalculateAllAccountStatuses...');
    System.debug('This method uses Database.update with allOrNone=false');
    System.debug('');
    
    // This method will:
    // 1. Query accounts with all needed fields
    // 2. Calculate statuses in memory
    // 3. Update using Database.update with allOrNone=false
    AccountTriggerHandler.recalculateAllAccountStatuses(accountIds);
    
    System.debug('✅ Status calculation completed!');
    System.debug('');
    
    // Verify results
    List<Account> results = [
        SELECT Name, Membership_Status__c, Registration_Status__c,
               Registration_Status_Flag__c, Survey_Total_Happy__c
        FROM Account 
        WHERE Id IN :accountIds
        LIMIT 10
    ];
    
    System.debug('========================================');
    System.debug('VERIFICATION - First 10 Accounts:');
    System.debug('========================================');
    
    Integer successCount = 0;
    
    for (Account acc : results) {
        Boolean hasStatuses = (acc.Membership_Status__c != null && acc.Registration_Status__c != null);
        
        System.debug('Account: ' + acc.Name);
        System.debug('  Membership Status: ' + acc.Membership_Status__c);
        System.debug('  Registration Status: ' + acc.Registration_Status__c);
        System.debug('  Survey Total Happy: ' + acc.Survey_Total_Happy__c);
        
        if (hasStatuses) {
            successCount++;
            System.debug('  ✅ Statuses populated!');
            
            // Verify logic
            if (acc.Survey_Total_Happy__c != null) {
                if (acc.Survey_Total_Happy__c >= 50) {
                    Boolean isCorrect = (acc.Registration_Status__c == 'Registered' && acc.Registration_Status_Flag__c == true);
                    System.debug('  ✅ Should be Registered: ' + (isCorrect ? 'CORRECT ✅' : 'INCORRECT ❌'));
                } else {
                    Boolean isCorrect = (acc.Registration_Status__c == 'Provisional' && acc.Registration_Status_Flag__c == false);
                    System.debug('  ✅ Should be Provisional: ' + (isCorrect ? 'CORRECT ✅' : 'INCORRECT ❌'));
                }
            }
        } else {
            System.debug('  ❌ Statuses still null');
        }
        System.debug('');
    }
    
    System.debug('========================================');
    System.debug('SUMMARY:');
    System.debug('  Accounts with statuses populated: ' + successCount + ' out of ' + results.size());
    System.debug('========================================');
}

System.debug('');
System.debug('========================================');
System.debug('SCRIPT COMPLETED');
System.debug('========================================');
